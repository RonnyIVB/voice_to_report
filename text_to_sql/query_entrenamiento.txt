-- TIPO: VISTA DE DIMENSIÓN (Dimension Table)
-- VISTA: dim_local
-- DESCRIPCIÓN: Un local puede ser una matriz de un negocio, punto de venta, o bodega.
CREATE VIEW dim_local AS (
    idlocal INT PRIMARY KEY, -- identificador del local
    empresa VARCHAR(255), -- nombre de la empresa a la que pertenece
    nombre VARCHAR(255) -- nombre de este local
);
---
-- TIPO: VISTA DE DIMENSIÓN (Dimension Table)
-- VISTA: dim_personal
CREATE VIEW dim_personal AS (
    idpersonal INT PRIMARY KEY, -- identificador del personal
    empresa VARCHAR(255), -- nombre de la empresa a la que pertenece
    nombre_completo VARCHAR(255) -- nombre completo del personal
);
---
-- TIPO: VISTA DE DIMENSIÓN (Dimension Table)
-- VISTA: dim_articulo
-- DESCRIPCIÓN: Un artículo es un bien o servicio que puede ser comprado, vendido o producido.
CREATE VIEW dim_articulo AS (
    idarticulo INT PRIMARY KEY, -- identificador del artículo
    empresa VARCHAR(255), -- nombre de la empresa a la que pertenece
    nombre VARCHAR(255), -- nombre del artículo
    tipo VARCHAR(100), -- indica el tipo del artículo, las opciones son muy variadas como 'ferretería', 'medicina', 'materia prima', etc.
    clasificacion VARCHAR(100), -- indica si el artículo es 'Mercadería', 'Activo Fijo' o 'Servicio'. Servicio puede ser de compra o de venta.
    precioventa NUMERIC(15,2), -- precio de venta del artículo
    preciocompra NUMERIC(15,2), -- precio de compra del artículo
    precioincluyeiva BOOLEAN, -- campo booleano que indica si el precio del artículo incluye IVA
    impuesto_nombre VARCHAR(100), -- Nombre del impuesto que tiene este artículo
    impuesto_porcentaje NUMERIC(5,2), -- Indica el porcentaje del impuesto de este artículo, si es cero entonces no tiene impuestos.
    activo BOOLEAN -- Indica si el producto está activo, true si está activo y false si es inactivo
);
---
-- TIPO: VISTA DE DIMENSIÓN (Dimension Table)
-- VISTA: dim_proveedor
CREATE VIEW dim_proveedor AS (
    idproveedor INT PRIMARY KEY, -- identificador del proveedor
    empresa VARCHAR(255), -- nombre de la empresa a la que pertenece
    nombre_completo VARCHAR(255), -- nombre completo del proveedor
    activo BOOLEAN -- campo booleano que indica que es true si el proveedor está activo
);
---
-- TIPO: VISTA DE DIMENSIÓN (Dimension Table)
-- VISTA: dim_cliente
CREATE VIEW dim_cliente AS (
    idcliente INT PRIMARY KEY, -- identificador del cliente
    empresa VARCHAR(255), -- nombre de la empresa a la que pertenece
    nombre_completo VARCHAR(255), -- nombre completo del cliente
    activo BOOLEAN -- campo booleano que indica que es true si el cliente está activo
);
---
-- TIPO: VISTA DE HECHOS (Fact Table)
-- VISTA: fact_porcobrar
-- GRANULARIDAD: Una fila representa un crédito a un cliente en una empresa específica. Que puede estar pendiente de pago o cancelado.
-- MÉTRICAS PRINCIPALES: valor_inicial, valor_pagado, valor_pendiente.
CREATE VIEW fact_porcobrar AS (
    idporcobrar INT PRIMARY KEY, -- identificador del documento por cobrar
    empresa VARCHAR(255), -- nombre de la empresa a la que pertenece
    idcliente INT REFERENCES dim_cliente(idcliente), -- Llave a la dimensión cliente
    estado VARCHAR(50), -- Indica el estado del documento, puede ser 'pendiente', 'cancelado' (que significa anulado) o 'vencido'
    numero VARCHAR(50), -- número del documento por cobrar
    fecha DATE, -- fecha en que se generó o registró el documento por cobrar
    fechainicio DATE, -- fecha en la que se deben iniciar los pagos
    fechafin DATE, -- fecha en la que se deben terminar los pagos
    valor_inicial NUMERIC(15,2), -- Valor o monto inicial de la deuda
    valor_pagado NUMERIC(15,2), -- Valor o monto que el cliente ya ha pagado
    valor_pendiente NUMERIC(15,2) -- Valor o monto que está pendiente de pago
);
---
-- TIPO: VISTA DE HECHOS (Fact Table)
-- VISTA: fact_porpagar
-- GRANULARIDAD: Una fila representa un crédito o deuda a un proveedor en una empresa específica. Que puede estar pendiente de pago o cancelado.
-- MÉTRICAS PRINCIPALES: valor_inicial, valor_pagado, valor_pendiente.
CREATE VIEW fact_porpagar AS (
    idporpagar INT PRIMARY KEY, -- identificador del documento por pagar
    empresa VARCHAR(255), -- nombre de la empresa a la que pertenece
    idproveedor INT REFERENCES dim_proveedor(idproveedor), -- Llave a la dimensión proveedor
    estado VARCHAR(50), -- Indica el estado del documento, puede ser 'pendiente', 'cancelado' (que significa anulado) o 'vencido'
    numero VARCHAR(50), -- número del documento por pagar
    fecha DATE, -- fecha en que se generó o registró el documento por pagar
    fechainicio DATE, -- fecha en la que se deben iniciar los pagos
    fechafin DATE, -- fecha en la que se deben terminar los pagos
    valor_inicial NUMERIC(15,2), -- Valor o monto inicial de la deuda
    valor_pagado NUMERIC(15,2), -- Valor o monto que ya se ha pagado al proveedor
    valor_pendiente NUMERIC(15,2) -- Valor o monto que está pendiente de pago al proveedor
);
---
-- TIPO: VISTA DE HECHOS (Fact Table)
-- VISTA: fact_factura
-- GRANULARIDAD: Una fila representa una venta a un cliente o compra a un proveedor de varios artículos en una empresa y local específicos.
-- MÉTRICAS PRINCIPALES: subtotal_impuestos, subtotal_sin_impuestos, impuestos y total.
CREATE VIEW fact_factura AS (
    idfactura INT PRIMARY KEY, -- identificador de la factura
    idlocal INT REFERENCES dim_local(idlocal), -- Llave a la dimensión local
    idproveedor INT REFERENCES dim_proveedor(idproveedor), -- Llave a la dimensión proveedor, distinto a null solo si es una compra
    idcliente INT REFERENCES dim_cliente(idcliente), -- Llave a la dimensión cliente, distinto a null solo si es una venta
    idpersonal INT REFERENCES dim_personal(idpersonal), -- Llave a la dimensión personal, es el identificador de la persona que registró la factura en el sistema
    idpersonalvendedor INT REFERENCES dim_personal(idpersonal), -- Llave a la dimensión personal, es el identificador del vendedor que suele recibir comisión.
    empresa VARCHAR(255), -- nombre de la empresa a la que pertenece
    estado VARCHAR(50), -- Estado de la factura, los valores pueden ser 'Cancelado', 'Anulada' y 'Pendiente'
    tipo_factura VARCHAR(100), -- El tipo de la factura, los valores pueden ser: 'Factura de Venta', 'Nota de Venta', 'Factura de Compra', 'Ingreso Nota de Venta', 'Liquidación de Compras', 'Proforma', 'Comprobante', 'Importación', 'Ingreso Proforma'
    tipo VARCHAR(50), -- Indica si es de compra o de venta, los valores son 'Compra' y 'Venta'
    forma_pago VARCHAR(50), -- Es la forma de pago, puede ser 'Contado', 'Crédito'
    fecha TIMESTAMP, -- fecha en la que se generó o registró la factura
    numero VARCHAR(50), -- es el número de la factura
    subtotal_impuestos NUMERIC(15,2), -- Es el monto o sumatoria de todos los artículos que en esta factura tienen impuesto
    subtotal_sin_impuestos NUMERIC(15,2), -- Es el monto o sumatoria de todos los artículos que en esta factura no tienen impuesto
    impuestos NUMERIC(15,2), -- es el valor o monto de los impuestos generados en esta factura
    total NUMERIC(15,2) -- Es el valor total de la factura
);
---
-- TIPO: VISTA DE HECHOS (Fact Table)
-- VISTA: fact_facturadetalle
-- GRANULARIDAD: Una fila representa un artículo de una factura en una empresa y local específicos. Una fila de fact_factura tiene una o varias filas de fact_facturadetalle.
-- MÉTRICAS PRINCIPALES: cantidad, precio_unitario, precio_total.
CREATE VIEW fact_facturadetalle AS (
    idfactura INT REFERENCES fact_factura(idfactura), -- Llave al hecho factura
    idarticulo INT REFERENCES dim_articulo(idarticulo), -- Llave a la dimensión artículo
    cantidad INT, -- La cantidad en unidades que se ha pedido del artículo
    precio_unitario NUMERIC(15,2), -- El precio unitario del producto sin impuestos en esta factura.
    precio_total NUMERIC(15,2), -- Indica el precio total del artículo sin impuestos, es la cantidad por el precio unitario
    impuestos VARCHAR(50) -- Indica si el artículo tiene impuestos, los valores pueden ser: null, 'iva'
);
