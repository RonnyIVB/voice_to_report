    valor_pagado NUMERIC(15,2), -- Valor o monto que el cliente ya ha pagado [cite: 13, 14]

-- TIPO: VISTA DE HECHOS (Fact Table) [cite: 19]
-- VISTA: fact_factura [cite: 19]
-- GRANULARIDAD: Una fila representa un venta a un cliente o compra a un proveedor de varios artículos en una empresa y local específicos. [cite: 19]
-- MÉTRICAS PRINCIPALES: subtotal_impuestos, subtotal_sin_impuestos, impuestos y total. [cite: 20]
CREATE VIEW fact_factura AS (
    idfactura INT PRIMARY KEY, -- identificador de la factura [cite: 20]
    idlocal INT REFERENCES dim_local(idlocal), -- Llave a la dimensión local [cite: 20]
    idproveedor INT REFERENCES dim_proveedor(idproveedor), -- Llave a la dimensión proveedor, distinto a null solo si es una compra [cite: 20]
    idcliente INT REFERENCES dim_cliente(idcliente), -- Llave a la dimensión, distinto a null solo si es una venta [cite: 20]
    idpersonal INT REFERENCES dim_personal(idpersonal), -- Llave a la dimensión personal, es el identificador de la persona que registró la factura en el sistema [cite: 20]
    idpersonalvendedor INT REFERENCES dim_personal(idpersonal), -- Llave a la dimensión personal, es el identificador del vendedor que suele recibir comisión. [cite: 20]
    empresa VARCHAR(255), -- nombre de la empresa a la que pertenece [cite: 21]
    estado VARCHAR(50), -- Estado de la factura, los valores pueden ser 'Cancelado', 'Anulada' y 'Pendiente' [cite: 21]
    tipo_factura VARCHAR(100), -- El tipo de la factura, los valores pueden ser: 'Factura de Venta', 'Nota de Venta', 'Factura de Compra', 'Ingreso Nota de Venta', 'Liquidación de Compras', 'Proforma', 'Comprobante', 'Importación', 'Ingreso Proforma' [cite: 21]
    tipo VARCHAR(50), -- Indica si es de compra o de venta, los valores son 'Compra' y 'Venta' [cite: 21]
    forma_pago VARCHAR(50), -- Es la forma de pago, puede ser 'Contado', 'Crédito' [cite: 21]
    fecha TIMESTAMP, -- fecha en la que se generó o registró la factura [cite: 21]
    numero VARCHAR(50), -- es el número de la factura [cite: 21, 22]
    subtotal_impuestos NUMERIC(15,2), -- Es el monto o sumatoria de todos los artículos que en esta factura no tienen impuesto [cite: 22]
    subtotal_sin_impuestos NUMERIC(15,2), -- Es el monto o sumatoria de todos los artículos que en esta factura sí tienen impuesto [cite: 22]
    impuestos NUMERIC(15,2), -- es el valor o monto de los impuestos generados en esta factura [cite: 22]
    total NUMERIC(15,2) -- Es el valor total de la factura [cite: 22]
);

-- TIPO: VISTA DE HECHOS (Fact Table) [cite: 24]
-- VISTA: fact_facturadetalle [cite: 24]
-- GRANULARIDAD: Una fila representa un artículo de una factura en una empresa y local específicos. Una fila de fact_factura tiene una o varias filas de fact_facturadetalle. [cite: 24, 25]
-- MÉTRICAS PRINCIPALES: cantidad, precio_unitario, precio_total. [cite: 25]
CREATE VIEW fact_facturadetalle AS (
    idfactura INT REFERENCES fact_factura(idfactura), -- Llave al hecho factura [cite: 26]
    idarticulo INT REFERENCES dim_articulo(idarticulo), -- Llave a la dimensión artículo [cite: 26]
    cantidad INT, -- La cantidad en unidades que se ha pedido del artículo [cite: 26]
    precio_unitario NUMERIC(15,2), -- El precio unitario del producto sin impuestos en esta factura. [cite: 26]
    precio_total NUMERIC(15,2), -- Indica el precio total del artículo sin impuestos, es la cantidad por el precio unitario [cite: 27]
    impuestos VARCHAR(50) -- Indica si el artículo tiene impuestos, los valores pueden ser: null, 'iva' [cite: 27]
);
